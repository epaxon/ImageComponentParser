1656         function run_segmentation(self)
1657             % run_segmentation(self): Runs the segmentation algorithm on
1658             % the ICs.
1659             %
1660             % This looks for ic components that are spatially localized.
1661             
1662             if self.data(self.gui.current_trial).stage < self.IcaStage
1663                 disp('Run ICA first');
1664                 return;
1665             end
1666             
1667             set(self.h.stage_status, 'String', 'Running Segmentation...');
1668             drawnow;
1669             
1670             thresh = self.data(self.gui.current_trial).settings.segment.thresh;
1671             down_size = 1./self.data(self.gui.current_trial).settings.segment.down_size;
1672             min_area = self.data(self.gui.current_trial).settings.segment.min_area;
1673             
1674             [segment_masks, segment_info] = segment_ics(self.data(self.gui.current_trial).ica.im_data, ...
1675                 thresh, down_size, min_area);
1676             
1677             % Make segment masks for each IC
1678             im_data = zeros(size(self.data(self.gui.current_trial).ica.im_data));
1679             for i = 1:size(self.data(self.gui.current_trial).ica.im_data, 3)
1680                 % Ok, make the im data all the masks for 1 segment.
1681                 idxs = find(segment_info.ic_ids == i);
1682                 if isempty(idxs)
1683                     % Then we didn't get anything out of a segment, which
1684                     % shouldn't happen...
1685                     disp(['No segments! ' num2str(i)]);
1686                     continue;
1687                 end
1688                 
1689                 im_sum = zeros(size(segment_masks(:,:,idxs(1))));
1690                 
1691                 pos_idxs = find(segment_info.is_pos(idxs));
1692                 neg_idxs = find(~segment_info.is_pos(idxs));
1693                 for j = 1:length(pos_idxs)
1694                     im_sum = im_sum + j * segment_masks(:,:,idxs(pos_idxs(j)));
1695                 end
1696                 for j = 1:length(neg_idxs)
1697                     im_sum = im_sum - j * segment_masks(:,:,idxs(neg_idxs(j)));
1698                 end
1699                 
1700                 im_data(:,:,i) = norm_range(im_sum);
1701             end
1702             
1703             % Now need to fix the rois
1704             im_x = self.data(self.gui.current_trial).ica.im_x;
1705             im_y = self.data(self.gui.current_trial).ica.im_y;
1706             segment_info.rois(:, 1) = interp1(1:size(segment_masks, 2), im_x, segment_info.rois(:, 1));
1707             segment_info.rois(:, 2) = interp1(1:size(segment_masks, 1), im_y, segment_info.rois(:, 2));
1708             segment_info.rois(:, 3) = mean(diff(im_x)) * segment_info.rois(:, 3) ./ sqrt(2);
1709             segment_info.rois(:, 4) = mean(diff(im_y)) * segment_info.rois(:, 4) ./ sqrt(2);
1710             
1711             self.data(self.gui.current_trial).segment.im_data = im_data;
1712             self.data(self.gui.current_trial).segment.im_x = self.data(self.gui.current_trial).ica.im_x;
1713             self.data(self.gui.current_trial).segment.im_y = self.data(self.gui.current_trial).ica.im_y;
1714             self.data(self.gui.current_trial).segment.segment_info = segment_info;
1715             
1716             self.create_new_roi_set('Segment_ROIs', true);
1717             self.data(self.gui.current_trial).rois{self.gui.d(self.gui.current_trial).current_roi_set} = segment_info.rois;
1718             
1719             self.h.roi_editor.set_rois(self.data(self.gui.current_trial).rois{self.gui.d(self.gui.current_trial).current_roi_set}, false);
1720             
1721             self.update();
1722             
1723             
1724             self.data(self.gui.current_trial).stage = self.PostProcessingStage;
1725             
1726             self.gui.d(self.gui.current_trial).last_display = 0;
1727             self.update();
